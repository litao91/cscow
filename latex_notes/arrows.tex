\section{Arrows}
\begin{longtable}{p{3cm} |p{5cm}| p{5cm}}\hline
    Implication & 
    \[ x^2=4 \implies x = \pm 2
    \]
    & \begin{verbatim}
x^2=4 \implies x = \pm 2
    \end{verbatim} \\\hline
    If and only if & 
    \[
        x^2=4 \iff x = \pm 2
    \] & 
    \begin{verbatim}
x^2=4 \iff x = \pm 2
    \end{verbatim}
    \\\hline
    Tends to &
    \[
        x \to +\infty
    \]
    &
    \begin{verbatim}
x \to +\infty
    \end{verbatim}\\\hline
    gets &
    \[
        A\gets B+C
    \]
    \[ 
        A \xleftarrow{\rm today} B
    \]
    \[
        B\xrightarrow{\rm tomorrow}C
    \]
    &
    \begin{verbatim}
A\gets B+C
A \xleftarrow{\rm today} B
B\xrightarrow{\rm tomorrow}C
    \end{verbatim}
    \\\hline
    Sizable single horizontal arrow &
    \[
    C \xrightarrow
    [\text{(exception Fridays)}]
    {\text{every day}} D
    \]&
    \begin{verbatim}
C \xrightarrow
[\text{(exception Fridays)}]
{\text{every day}} D
\end{verbatim} \\\hline
 Sizeable single vertical arrows &
\[
\left\uparrow\sum \right\downarrow\;
\Big\updownarrow
\]
&
\begin{verbatim}
\left\uparrow\sum \right\downarrow\;
\Big\updownarrow
\end{verbatim}\\\hline

    \left\Uparrow\sum \right\Downarrow \;
    \Big\Updownarrow
\]

\left\Uparrow\sum \right\Downarrow \;
\Big\Updownarrow
\end{verbatim}
