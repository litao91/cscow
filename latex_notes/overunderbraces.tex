\section{Over and underbraces and other embellishments}
\begin{longtable}{p{3cm} |p{5cm}| p{5cm}}\hline
    overline & 
\[
    \overline{A+B}
\]
&
\begin{verbatim}
\overline{A+B}
\end{verbatim} \\\hline
underline &
\[
\underline{A+B}
\] &
\begin{verbatim}
\underline{A+B}
\end{verbatim}
\\\hline
Hat &
\[
    \widehat{A+B}
\]
\begin{verbatim}
\widehat{A+B}
\end{verbatim}
\\\hline
Tilde&
\[
    \widetilde{A+B}
\]
&
\begin{verbatim}
\widetilde{A+B}
\end{verbatim}\\\hline
Vector markers &
\[
\overrightarrow{AB}\text{ and }
\overleftarrow{BA}
\] &
\begin{verbatim}
\overrightarrow{AB}\text{ and }
\overleftarrow{BA}
\end{verbatim}\\\hline'
Overbrace &
\[
\overbrace{x_1+x_2+\cdots + x_k}^
{k \text{ in total}}
\] &
\begin{verbatim}
\overbrace(x_1+x_2+\cdots + x_k}^
{k \text{ in total}}
\end{verbatim}
\\\hline
Underbrace &
\[
    m^n = \underbrace{m\cdot m\cdots m}_{n}
\] &
\begin{verbatim}
    m^n = \underbrace{m\cdot m\cdots m}_{n}
\end{verbatim}\\\hline
Affixing arbitrary symbols &
\[
x\overset{?}{\ge}y
\]
&
\begin{verbatim}
x\overset{?}{\ge}y
\end{verbatim}
\end{longtable}
